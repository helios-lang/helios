Koi Grammar

program ::= top-level-listopt EOF ;
top-level-list ::= top-level
  | top-level NEWLINE top-level-list ;
 
top-level ::= declaration
  | expression ;
 
declaration ::= function-declaration
  | module-declaration
  | type-declaration
  | using-declaration ;
expression ::= if-expression
  | let-expression
  | literal-expression
  | match-expression ;
 
function-declaration ::= def visibility-modifieropt IDENTIFIER ( parameter-listopt ) type-annotationopt = expression-block ;
module-declaration ::= module visibility-modifieropt IDENTIFIER = module-declaration-block ;
type-declaration ::= type visibility-modifieropt IDENTIFIER = type-declaration-block ;
using-declaration ::= using PATH using-declaration-aliasopt
  | using PATH using-declaration-importopt ;
 
visibility-modifier ::= public
  | internal ;
 
parameter ::= IDENTIFIER type-annotationopt ;
parameter-list ::= parameter
  | parameter , parameter-list ;
module-declaration-block ::= INDENT module-declaration-block-item OUTDENT ;
type-declaration-block ::= enum-body
  | record-body ;
 
using-declaration-import ::= . IDENTIFIER using-declaration-aliasopt
  | . { using-declaration-import-listopt } ;
using-declaration-import-list ::= IDENTIFIER using-declaration-aliasopt using-declaration-import-restopt ,opt ;
using-declaration-import-rest ::= , IDENTIFIER using-declaration-aliasopt ;
using-declaration-alias ::= as IDENTIFIER ;
 
module-declaration-block-item ::= function-declaration
  | module-declaration
  | type-declaration ;
 
enum-body ::= |opt IDENTIFIER enum-body-restopt ;
enum-body-rest ::= | IDENTIFIER enum-body-restopt ;
record-body ::= { record-body-fieldsopt } ;
record-body-fields ::= IDENTIFIER type-annotation record-body-restopt ,opt ;
record-body-rest ::= , IDENTIFIER type-annotation record-body-restopt ;
 
type ::= array-type
  | function-type
  | tuple-type
  | IDENTIFIER ;
array-type ::= [ type ] ;
function-type ::= type -> type
  | type -> function-type ;
tuple-type ::= tuple-type-list
  | ( tuple-type-listopt ) ;
tuple-type-list ::= type
  | type , tuple-type-list ;
type-annotation ::= : type ;
 
if-expression ::= if pattern then expression-block else-clauseopt ;
else-clause ::= else expression-block
  | else if-expression ;
 
let-expression ::= let pattern = expression-block ;
 
match-expression ::= match pattern with match-expression-clause ;
match-expression-clause ::= |opt pattern -> expression-block match-expression-clause-restopt ;
match-expression-clause-rest ::= | pattern -> expression-block match-expression-clause-restopt ;
 
expression-block ::= expression
  | INDENT expression-block-list OUTDENT ;
expression-block-list ::= expression
  | expression ; expression-block-list
  | expression NEWLINE expression-block-list ;
